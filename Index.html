  <script>

    // Target date at local midnight on 3 Oct 2033.

    // If you want Kuwait time specifically, use the second option below.

    const target = new Date(2033, 9, 3); // months are 0-based: 9 = October


    // For a fixed Kuwait time (UTC+3, no DST), uncomment the next line and comment the one above:

    // const target = new Date('2033-10-03T00:00:00+03:00');


    function daysUntil(t) {

      const now = new Date();

      // Start-of-day (local) to show whole days remaining and update cleanly at midnight

      const todayStart = new Date(now.getFullYear(), now.getMonth(), now.getDate());

      const msPerDay = 24 * 60 * 60 * 1000;

      const diffDays = Math.max(0, Math.ceil((t - todayStart) / msPerDay));

      return diffDays;

    }


    function render() {

      document.getElementById('days').textContent = daysUntil(target);

    }


    render();


    // Re-render just after local midnight to keep the number accurate

    (function scheduleMidnightTick() {

      const now = new Date();

      const next = new Date(now.getFullYear(), now.getMonth(), now.getDate() + 1, 0, 0, 1);

      setTimeout(() => { render(); scheduleMidnightTick(); }, next - now);

    })();

  </script>
